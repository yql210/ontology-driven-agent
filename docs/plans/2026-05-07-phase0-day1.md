# src/layerkg/schema.py
from __future__ import annotations

import uuid
from dataclasses import dataclass, field
from datetime import datetime, timezone


VALID_CODE_TYPES = frozenset({"function", "class", "interface", "module", "file"})
VALID_CONCEPT_TYPES = frozenset({"business_concept", "design_pattern", "api_contract", "data_model", "process"})
VALID_DOC_TYPES = frozenset({"readme", "module_doc", "api_doc", "comment", "wiki", "architecture_doc"})
VALID_RESOURCE_TYPES = frozenset({"image", "diagram", "pdf", "config", "schema_file", "log"})


@dataclass
class CodeEntity:
    """代码实体：函数/类/接口/模块/文件。"""

    name: str
    entity_type: str
    id: str = field(default_factory=lambda: str(uuid.uuid4()))
    file_path: str | None = None
    start_line: int | None = None
    end_line: int | None = None
    source: str | None = None
    language: str | None = None
    created_at: str = field(default_factory=lambda: datetime.now(timezone.utc).isoformat())

    def __post_init__(self) -> None:
        if self.entity_type not in VALID_CODE_TYPES:
            raise ValueError(
                f"entity_type must be one of {VALID_CODE_TYPES}, got '{self.entity_type}'"
            )
```

# src/layerkg/graph_store.py
from __future__ import annotations

from abc import ABC, abstractmethod

from layerkg.schema import CodeEntity, Relation


class GraphStore(ABC):
    """图存储抽象基类，定义统一的 CRUD 接口。

    具体实现（Neo4j、内存等）继承此类。
    """

    @abstractmethod
    def merge_node(self, entity: CodeEntity | object) -> None:
        """创建或更新节点。"""

    @abstractmethod
    def get_node(self, entity_id: str) -> object | None:
        """根据 ID 获取节点，不存在返回 None。"""

    @abstractmethod
    def delete_node(self, entity_id: str) -> bool:
        """删除节点，成功返回 True。"""

    @abstractmethod
    def merge_relation(self, relation: Relation) -> None:
        """创建或更新关系。"""

    @abstractmethod
    def get_relations(
        self,
        source_id: str | None = None,
        target_id: str | None = None,
        relation_type: str | None = None,
    ) -> list[Relation]:
        """查询关系，支持按源/目标/类型过滤。"""

    @abstractmethod
    def query(self, cypher: str, params: dict | None = None) -> list[dict]:
        """执行原始 Cypher 查询。"""
```

import os
from dataclasses import dataclass


@dataclass
class LayerKGConfig:
    """LayerKG 配置。"""

    neo4j_uri: str = "bolt://localhost:7687"
    neo4j_user: str = "neo4j"
    neo4j_password: str = ""

        return cls(
            neo4j_uri=os.getenv("NEO4J_URI", cls.neo4j_uri),
            neo4j_user=os.getenv("NEO4J_USER", cls.neo4j_user),
